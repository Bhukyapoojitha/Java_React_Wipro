# File Path: src/api/axiosConfig.js
import axios from 'axios';

const API_BASE_URL = 'http://localhost:8080/api';

const axiosInstance = axios.create({
 baseURL: API_BASE_URL,
 headers: {
   'Content-Type': 'application/json',
 },
});

axiosInstance.interceptors.request.use(
 (config) => {
   const token = localStorage.getItem('token');
   if (token) {
     config.headers['Authorization'] = `Bearer ${token}`;
   }
   return config;
 },
 (error) => Promise.reject(error)
);

axiosInstance.interceptors.response.use(
 (response) => response,
 (error) => {
   if (error.response && error.response.status === 401) {
     localStorage.removeItem('token');
     window.location.href = '/login';
   }
   return Promise.reject(error);
 }
);

export default axiosInstance;

import React, { createContext, useContext, useState, useEffect } from 'react';
import { tokenUtils } from '../utils/tokenUtils';

const AuthContext = createContext(null);

export const AuthProvider = ({ children }) => {
 const [user, setUser] = useState(null);
 const [token, setToken] = useState(localStorage.getItem('token') || null);
 const [loading, setLoading] = useState(true);

 useEffect(() => {
   if (token) {
     localStorage.setItem('token', token);
     const profile = tokenUtils.decodeTokenClaims(token);
     setUser(profile);
   } else {
     localStorage.removeItem('token');
     setUser(null);
   }
   setLoading(false);
 }, [token]);

 const loginSession = (accessToken) => setToken(accessToken);
 const logoutSession = () => setToken(null);

 return (
   <AuthContext.Provider value={{ user, token, loading, loginSession, logoutSession }}>
     {children}
   </AuthContext.Provider>
 );
};

export const useAuth = () => useContext(AuthContext);

export default function QuestionDetails() {

 const navigate = useNavigate();
 const [question, setQuestion] = useState(null);
 const [answers, setAnswers] = useState([]);
 const [loading, setLoading] = useState(true);
 const [aiLoading, setAiLoading] = useState(false);
 const [error, setError] = useState('');

 useEffect(() => {
   initData();
 }, [id]);

 const initData = async () => {
   try {
     const qData = await questionApi.fetchById(id);
     setQuestion(qData);
     const aData = await answerApi.getAnswersForQuestion(id);
     setAnswers(aData);
   } catch { setError('Error syncing data models tracking values.'); }
   finally { setLoading(false); }
 };

 const handleAnswerSubmit = async (e) => {
   e.preventDefault();
   const txt = e.target.answerText.value;
   if (!txt.trim()) return;
   try {
     await answerApi.postAnswerRecord(id, txt);
     e.target.answerText.value = '';
     const updated = await answerApi.getAnswersForQuestion(id);
     setAnswers(updated);
   } catch { setError('Failed to register tracking resolution block.'); }
 };

 const queryAiSuggestionModel = async () => {
   setAiLoading(true); setError('');
   try {
     const suggestion = await aiApi.fetchSolutionSuggestion(id);
     document.getElementById('answerText').value = suggestion.payload;
   } catch { setError('AI core pipeline iteration limits boundary timeout.'); }
   finally { setAiLoading(false); }
 };

 if (loading) return <Loader />;
 if (!question) return <div className="text-gray-400 text-sm">Thread structural index mismatched or deleted.</div>;

export const tokenUtils = {
 decodeTokenClaims: (token) => {
   try {
     const base64Url = token.split('.')[1];
     const base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
     const jsonPayload = decodeURIComponent(
       window.atob(base64)
         .split('')
         .map((c) => '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2))
         .join('')
     );
     const claims = JSON.parse(jsonPayload);
     return {
       email: claims.sub,
       role: claims.role || (claims.authorities ? claims.authorities[0].replace('ROLE_', '') : 'USER'),
       name: claims.name || claims.sub.split('@')[0]
     };
   } catch {
     return null;
   }
 }
};
